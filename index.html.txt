    <script>
        // Aquí defines tus "escenas"
        const escenas = {
            inicio: {
                texto: "Te despiertas en un bosque desconocido. ¿Qué haces?",
                opciones: [
                    { texto: "Caminar hacia el río", siguiente: "rio" },
                    { texto: "Subir a un árbol", siguiente: "arbol" }
                ]
            },
            rio: {
                texto: "Llegas al río y ves una balsa. ¿Cruzas o sigues la orilla?",
                opciones: [
                    { texto: "Cruzar", siguiente: "final_bueno" },
                    { texto: "Seguir la orilla", siguiente: "inicio" }
                ]
            },
            arbol: {
                texto: "Desde arriba ves una cabaña humeante. ¿Bajas a investigar?",
                opciones: [
                    { texto: "Ir a la cabaña", siguiente: "final_misterioso" },
                    { texto: "Volver al inicio", siguiente: "inicio" }
                ]
            },
            final_bueno: {
                texto: "¡Felicidades! La balsa te llevó a un pueblo seguro.",
                opciones: [{ texto: "Reiniciar", siguiente: "inicio" }]
            },
            final_misterioso: {
                texto: "La cabaña estaba llena de pasteles. Te quedaste a vivir allí.",
                opciones: [{ texto: "Reiniciar", siguiente: "inicio" }]
            }
        };

        function mostrarEscena(nombreEscena) {
            const escena = escenas[nombreEscena];
            document.getElementById('texto-pregunta').innerText = escena.texto;
            const contenedorBotones = document.getElementById('botones');
            
            // Limpiamos botones anteriores
            contenedorBotones.innerHTML = '';

            // Creamos los nuevos botones
            escena.opciones.forEach(opcion => {
                const boton = document.createElement('button');
                boton.innerText = opcion.texto;
                boton.onclick = () => mostrarEscena(opcion.siguiente);
                contenedorBotones.appendChild(boton);
            });
        }

        // Iniciar el juego
        mostrarEscena('inicio');
    </script>
